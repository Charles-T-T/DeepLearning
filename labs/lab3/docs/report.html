class Softmax:
    def __init__(self, device="cpu"):
        self.device = device


    def forward(self, x):
        x_max = torch.max(x, dim=1, keepdim=True) # é˜²æ­¢æº¢å‡ºï¼Œå‡å»æœ€å¤§å€¼æ¥å¢åŠ æ•°å€¼ç¨³å®šæ€§
        x_exp = torch.exp(x - x_max.values)
        self.probs = x_exp / torch.sum(x_exp, dim=1, keepdim=True)
        return self.probs
    
    def cross_entropy(self, predictions, truths):
        probs = self.probs  # Softmaxå¾—åˆ°çš„é¢„æµ‹æ ‡ç­¾çš„æ¦‚ç‡åˆ†å¸ƒ


        # è®¾å®šä¸€ä¸ªæœ€å°å€¼min_valï¼Œé˜²æ­¢log(0)

        self.truth_onehot = truth_onehot  # çœŸå®æ ‡ç­¾çš„one-hotåˆ†å¸ƒ

class MLP:

def train_and_evaluate(model, train_loader, test_loader, epochs, learning_rate):
    train_loss_history = []
    test_loss_history = []
    
    for epoch in range(epochs):
        total_loss = 0
        for images, labels in train_loader:
            probs = model.forward(images)  # å‰å‘ä¼ æ’­
            loss = model.softmax.cross_entropy(probs, labels)  # è®¡ç®—æŸå¤±
            model.backward()  # åå‘ä¼ æ’­
            model.update(learning_rate)  # æ›´æ–°å‚æ•°
            total_loss += loss.item()
        
        # è®¡ç®—å¹³å‡è®­ç»ƒæŸå¤±
        avg_train_loss = total_loss / len(train_loader)
        train_loss_history.append(avg_train_loss)
        
        # è¯„ä¼°æ¨¡å‹åœ¨æµ‹è¯•é›†ä¸Šçš„å‡†ç¡®ç‡
        accuracy, avg_test_loss = evaluate(model, test_loader)
        test_loss_history.append(avg_test_loss)


        print(f"Epoch {epoch+1}/{epochs}, "
              f"Train Loss: {avg_train_loss:.4f}, "
              f"Test Accuracy: {(accuracy * 100):.4f}%")
        
    # ç»˜åˆ¶æ›²çº¿
    plt.figure(figsize=(10, 6))
    plt.plot(range(epochs), train_loss_history, label='Train Loss')
    plt.plot(range(epochs), test_loss_history, label='Test Loss')
    plt.xlabel('Epoch')
    plt.ylabel('Loss')
    plt.title('Training and Test Loss Curve')
    plt.legend()
    plt.grid(True)

def evaluate(model, test_loader):
    correct = 0
    total = 0
    total_test_loss = 0
    
    for images, labels in test_loader:
        probs = model.forward(images)
        loss = model.softmax.cross_entropy(probs, labels)
        total_test_loss += loss.item()
        
        # è·å–é¢„æµ‹çš„ç±»åˆ«
        _, predicted = torch.max(probs, dim=1)  # è¾“å‡ºæ¯è¡Œæœ€å¤§å€¼çš„ç´¢å¼•ï¼Œå³é¢„æµ‹çš„ç±»åˆ«
        
        # ç»Ÿè®¡æ­£ç¡®é¢„æµ‹çš„æ•°é‡
        correct += (predicted == labels).sum().item()

    accuracy = correct / total  # è®¡ç®—å‡†ç¡®ç‡
    avg_test_loss = total_test_loss / len(test_loader)
    
    return accuracy, avg_test_loss
